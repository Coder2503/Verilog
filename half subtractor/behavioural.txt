//using operators
module hs(d,bo,a,b);
	input a,b;
	output reg d,bo;
	always@(a or b)
	begin
		d=a^b;
		bo=~a&b;
	end
endmodule


//using case statement
module hs(d,bo,a,b);
	input a,b;
	output reg d,bo;
	always@(a or b)
	begin
		case({a,b})
		2'b00:begin d=0;bo=0; end
		2'b01:begin d=1;bo=1; end
		2'b10:begin d=1;bo=0; end
		2'b11:begin d=0;bo=0; end
		default:begin d=0;bo=0; end
		endcase
	end
endmodule
