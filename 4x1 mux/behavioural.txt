//using operators
module mux4x1(Y,I,S);
	input [3:0]I;
	input [1:0]S;
	output reg Y;
	always@(*)
	begin
		assign Y=S[1]?(S[0]?I[3]:I[2]):(S[0]?I[1]:I[0]);
	end
endmodule


//using case statement
module mux4x1(Y,I,S);
	input [3:0]I;
	input [1:0]S;
	output reg Y;
	always@(*)
	begin
		case(S)
		2'b00:Y=I[0];
		2'b01:Y=I[1];
		2'b10:Y=I[2];
		2'b11:Y=I[3];
		default:Y=0;
		endcase
	end
endmodule
