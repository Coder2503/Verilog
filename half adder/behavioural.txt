//using operators
module ha(s,cy,a,b);
	input a,b;
	output reg s,cy;
	always@(a or b)
	begin
		s=a^b;
		cy=a&b;
	end
endmodule


//using case statement
module ha(s,cy,a,b);
	input a,b;
	output reg s,cy;
	always@(a or b)
	begin
		case({a,b})
		2'b00:begin s=0;cy=0; end
		2'b01:begin s=1;cy=0; end
		2'b10:begin s=1;cy=0; end
		2'b11:begin s=0;cy=1; end
		default:begin s=0;cy=0; end
		endcase
	end
endmodule
