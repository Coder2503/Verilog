module mod6(Q,led,clk,clear);
	input clk,clear;
	output reg [2:0]Q;
	output reg led=0;
	reg [25:0]cnt=25'd0;
	always@(posedge clk)
	begin
		cnt<=cnt+1;
		if(cnt==25'd12000000)
		begin
			led=~led;
			cnt<=0;
		end
	end
	always@(posedge led)
	begin
		if(clear|Q=3'd6)
		Q<=0;
		else
		Q<=Q+1;
	end
endmodule
